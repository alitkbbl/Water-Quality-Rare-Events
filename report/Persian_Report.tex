	\section{مقدمه و بیان مسئله \lr{(Introduction \& Problem Statement)}}
	
	در سیستم‌های پایش کیفیت آب، حفظ پارامترهای شیمیایی در محدوده‌های استاندارد برای تضمین سلامت عمومی از اهمیت حیاتی برخوردار است. انحراف متغیرهای کلیدی، مانند $pH$ یا غلظت مواد شیمیایی، از آستانه‌های مجاز می‌تواند به خطرات جدی بهداشتی منجر شود. از منظر آماری، از آنجا که سیستم‌های تصفیه آب معمولاً با دقت بالایی کار می‌کنند، خروج این پارامترها از حد استاندارد بسیار کم‌رخداد بوده و در دسته \textbf{رویدادهای نادر \lr{(Rare Events)}} طبقه‌بندی می‌شود.
	
	هدف اصلی این پروژه، مدل‌سازی ریاضی یکی از پارامترهای کیفیت آب و تخمین دقیق احتمال وقوع این رویدادهای نادر در دم‌های توزیع \lr{(Tail Distributions)} است؛ یعنی تخمین احتمالی به فرم
	\[
	\gamma = P(X > \tau)
	\]
	که در آن $\tau$ آستانه بحرانی است. از آنجایی که در رویدادهایی با احتمال بسیار پایین، مثلاً $\gamma \approx 10^{-4}$، روش‌های نمونه‌گیری ساده \lr{(Standard Monte Carlo)} دچار واریانس بالا و هزینه محاسباتی شدید می‌شوند، این پروژه در فازهای بعدی به سراغ پیاده‌سازی تکنیک‌های پیشرفته کاهش واریانس نظیر \textbf{نمونه‌گیری اهمیتی \lr{(Importance Sampling)}} خواهد رفت. گام نخست این مسیر، شناخت دقیق داده‌ها و یافتن بهترین توزیع احتمال پایه $f(x)$ برای مدل‌سازی است که در این بخش به آن پرداخته می‌شود.

	\section{توصیف داده‌ها و تحلیل آماری \lr{(Exploratory Data Analysis)}}
	
	در گام نخست پروژه، مجموعه داده کیفیت آب \lr{water\_potability.csv} شامل ۳۲۷۶ نمونه مورد بررسی قرار گرفت. با توجه به وجود مقادیر گمشده \lr{(Missing Values)} در پارامترهایی مانند $pH$ و لزوم داشتن یک مجموعه داده کامل برای برازش دقیق توزیع، پارامتر \textbf{کلرامین \lr{(Chloramines)}} به عنوان متغیر هدف انتخاب شد. کلرامین یک پارامتر پیوسته شیمیایی است که هیچ‌گونه داده گمشده‌ای ندارد و افزایش آن از حد مجاز، خطری برای سلامت عمومی و یک رویداد نادر در این سیستم محسوب می‌شود.
	
	بررسی‌های آماری نشان می‌دهد که میانگین غلظت کلرامین در داده‌ها $\mu \approx 7.12$ و انحراف معیار آن $\sigma \approx 1.58$ است. نکته بسیار مهم در تحلیل توصیفی این متغیر، میزان چولگی \lr{(Skewness)} آن است که تقریباً برابر با صفر، یعنی $-0.012$، به دست آمد. این عدد نشان‌دهنده تقارن بالای داده‌ها و شباهت رفتاری آن به توزیع‌های متقارن زنگوله‌ای شکل است. با این حال، همان‌طور که در نمودار جعبه‌ای \lr{(Box-plot)} مشاهده می‌شود، داده‌های پرت \lr{(Outliers)} در هر دو سمت دم توزیع، پایین‌تر از ۳ و بالاتر از ۱۱، وجود دارند که دقیقاً همان نواحی بحرانی ما برای بررسی رویدادهای نادر را تشکیل می‌دهند.

	\section{مدل‌سازی ریاضی و برازش توزیع \lr{(Distribution Fitting)}}
	
	برای انجام شبیه‌سازی‌های مونت‌کارلو در فازهای بعدی، نیازمند یافتن بهترین تابع چگالی احتمال نظری $f(x)$ هستیم که رفتار داده‌های کلرامین را نمایندگی کند. برای این منظور، سه توزیع \textbf{نرمال \lr{(Normal)}}، \textbf{گاما \lr{(Gamma)}} و \textbf{وایبول \lr{(Weibull)}} در نظر گرفته شدند.
	
	جهت اطمینان از صحت محاسبات، برآورد پارامترهای این توزیع‌ها به دو روش انجام شد: پیاده‌سازی دستی الگوریتم \textbf{برآورد حداکثر درستی \lr{(Manual MLE)}} و استفاده از توابع بهینه‌شده کتابخانه \lr{scipy}. تطابق کامل خروجی‌های این دو روش، به عنوان مثال $\mu=7.1223$ و $\sigma=1.5828$ برای توزیع نرمال در هر دو حالت، صحت پیاده‌سازی الگوریتم‌ها را تایید می‌کند.
	
	برای ارزیابی و انتخاب بهترین مدل، از معیارهای لگاریتم درست‌نمایی \lr{(Log-likelihood)}، معیار اطلاعات آکائیکه \lr{(AIC)}، معیار اطلاعات بیزی \lr{(BIC)} و فاصله در آزمون کولموگوروف-اسمیرنف \lr{(KS)} استفاده شد. نتایج در جدول زیر خلاصه شده است:

	\textbf{تحلیل نتایج برازش:}
	همان‌طور که از معیارهای ارزیابی مشخص است، \textbf{توزیع نرمال} با اختلاف معناداری بهترین عملکرد را دارد. کمترین میزان \lr{AIC} و \lr{BIC} نشان می‌دهد که توزیع نرمال با کمترین پیچیدگی، بیشترین میزان اطلاعات داده‌ها را حفظ کرده است. همچنین کمترین خطای \lr{KS}، یعنی $0.021$، نشان‌دهنده بالاترین تطابق میان تابع توزیع تجمعی تجربی \lr{(ECDF)} و مدل نظری است.
	
	دلیل ریاضی این برتری به ماهیت متقارن داده‌های کلرامین، یعنی چولگی نزدیک به صفر، بازمی‌گردد؛ در حالی که توزیع‌های گاما و وایبول معمولاً برای مدل‌سازی داده‌های دارای چولگی \lr{(Skewed Data)} عملکرد بهتری از خود نشان می‌دهند. نمودار \lr{Q-Q} نیز این نتیجه را تایید می‌کند؛ نقاط تجربی به خوبی روی خط نیم‌ساز قرار گرفته‌اند و تنها در انتهای دم راست \lr{(Extreme Right Tail)} انحراف بسیار جزئی دیده می‌شود که در نمونه‌گیری‌های محدود طبیعی است.

	\section{تعریف آستانه رویداد نادر \lr{(Rare-Event Threshold)}}
	
	مدل کاری نهایی ما برای ادامه پروژه، توزیع $\operatorname{Normal}(\mu=7.1223, \sigma=1.5828)$ است. هدف شبیه‌سازی، تخمین احتمال وقوع مقادیر بحرانی کلرامین است. برای پیش‌برد فاز محاسباتی، آستانه بحرانی $\tau$ به گونه‌ای تنظیم شد که احتمال عبور از آن در مدل نظری، یک رویداد نادر با احتمال دقیقاً $\gamma = 10^{-4}$ باشد.
	
	بر اساس محاسبات تابع بقا \lr{(Survival Function)} روی مدل نرمال، این آستانه برابر با \textbf{$\tau \approx 13.0089$} محاسبه گردید. با نگاهی به داده‌های تجربی، مشاهده می‌شود که حداکثر مقدار ثبت شده در کل مجموعه داده برابر با \textbf{$13.1270$} است. این بدین معناست که آستانه $\tau$ دقیقاً در لبه انتهایی توزیع تجربی قرار گرفته است.
	
	از منظر شبیه‌سازی، نادر بودن این رویداد چالش‌برانگیز است؛ زیرا اگر بخواهیم با نمونه‌گیری مونت‌کارلوی خام \lr{(Standard MC)} احتمال مقادیر بزرگتر از $13.0089$ را با خطای کم تخمین بزنیم، به میلیون‌ها نمونه نیاز خواهیم داشت که در نوت‌بوک‌های بعدی این ناکارآمدی اثبات شده و با استفاده از \lr{Importance Sampling} برطرف خواهد شد.
	
	\section{شبیه‌سازی و تولید داده‌های مصنوعی \lr{(Accept-Reject Sampling)}}
	
	همان‌طور که پیش‌تر اشاره شد، تحلیل رویدادهای نادر با احتمالاتی نظیر $\gamma = 10^{-4}$ نیازمند حجم عظیمی از داده‌ها است. از آنجا که مجموعه داده واقعی تنها شامل ۳۲۷۶ مشاهده است، در این فاز از پروژه باید به کمک مدل برازش شده، یعنی توزیع نرمال با $\mu=7.1223$ و $\sigma=1.5828$، نمونه‌های جدیدی تولید کنیم. با توجه به محدودیت استفاده از توابع آماده تولید اعداد تصادفی، مانند \lr{scipy.stats.norm.rvs}، الگوریتم پایه و بنیادین \textbf{پذیرش-رد \lr{(Accept-Reject)}} به صورت دستی پیاده‌سازی گردید.
	
	\textbf{طراحی الگوریتم و انتخاب توزیع پیشنهادی:}
	در این روش، هدف استخراج نمونه از توزیع هدف $f(x)$ به کمک یک توزیع پیشنهادی \lr{(Proposal Distribution)} ساده‌تر مانند $g(x)$ است؛ به شرطی که نامساوی زیر همواره برقرار باشد:
	\[
	f(x) \le M \cdot g(x)
	\]
	در این پروژه، توزیع پیشنهادی از نوع \textbf{یکنواخت \lr{(Uniform)}} در بازه $[a,b]$ انتخاب شد. انتخاب این بازه نیازمند یک تحلیل دقیق است:
	
	\begin{itemize}
		\item \textbf{کران پایین ($a = 0$):} با توجه به ماهیت فیزیکی متغیر کلرامین، یعنی غلظت مواد شیمیایی، مقادیر منفی بی‌معنی هستند.
		\item \textbf{کران بالا ($b = \mu + 6\sigma \approx 16.62$):} در توزیع نرمال، احتمال حضور داده‌ها فراتر از فاصله شش انحراف معیار از میانگین، کمتر از $10^{-8}$ است. بنابراین، برش \lr{(Truncation)} توزیع در این نقطه، خطای محاسباتی کاملاً قابل چشم‌پوشی ایجاد می‌کند.
	\end{itemize}
	
	با جایگذاری توزیع یکنواخت $g(x) = \frac{1}{b-a}$، ثابت پوششی به صورت
	\[
	M = (b-a)\max f(x)
	\]
	محاسبه می‌شود که مقدار آن در این مسئله \textbf{$M = 4.1891$} به دست آمد.
	
	\textbf{صحت‌سنجی \lr{(Validation)} خروجی‌ها:}
	با استفاده از این رویکرد، ۱۰۰,۰۰۰ نمونه تولید شد. برای تایید صحت نمونه‌های تولید شده، سه معیار در نظر گرفته شد:
	
\begin{enumerate}
	\item \textbf{گشتاورها:} میانگین نمونه‌های تولید شده، یعنی $7.1221$، و انحراف معیار آن‌ها، یعنی $1.5839$، تطابق تقریباً کاملی با پارامترهای تئوری دارند.
	\item \textbf{آزمون آماری \lr{KS}:} نتیجه آزمون کولموگوروف-اسمیرنف با مقدار \lr{p-value} برابر با $0.932$ نشان می‌دهد که هیچ شواهد آماری برای رد این فرضیه که نمونه‌ها از توزیع $f(x)$ آمده‌اند، وجود ندارد. به عبارتی، تطابق قوی ای است.
	\item \textbf{بررسی بصری:} مقایسه هیستوگرام داده‌های واقعی و نمونه‌های تولید شده، میزان انطباق الگوریتم را نشان می‌دهد.
\end{enumerate}

	\section{تحلیل کارایی: بررسی اثر ثابت پوششی $M$ بر نرخ پذیرش و زمان اجرا}
	
	\textbf{پاسخ به پرسش تحلیلی اول \lr{(Analytical Question 1)}}\label{sec:q1}
	
	الگوریتم پذیرش-رد، علیرغم سادگی و دقت، می‌تواند از منظر محاسباتی به شدت ناکارآمد باشد. دلیل این امر در متغیر تصادفی $U$ و شرط پذیرش زیر نهفته است:
	\[
	U \le \frac{f(Y)}{M \cdot g(Y)}
	\]
	از نظر ریاضی، امید ریاضی \lr{(Expected Value)} نرخ پذیرش در این الگوریتم دقیقاً برابر با $\frac{1}{M}$ است. به عبارت دیگر، هرچه ثابت $M$ بزرگتر باشد، فضای مرده \lr{(Dead Space)} بین توزیع پوششی و توزیع هدف بیشتر شده و نمونه‌های بیشتری هدر می‌روند یا \lr{Reject} می‌شوند.
	
	برای بررسی تجربی این پدیده، الگوریتم با عرض‌های مختلفی برای توزیع پیشنهادی یکنواخت، از شعاع $3\sigma$ تا $20\sigma$، اجرا شد. نتایج به دست آمده در جدول و نمودارهای زیر، حاوی دو تحلیل کلیدی است:
	
	\begin{figure}[H]
		\centering
		\includegraphics[width=0.95\textwidth]{../results/02_acceptance_vs_M.png}
		\caption{سمت چپ: کاهش نرخ پذیرش با افزایش ثابت $M$ و تطابق دقیق رفتار تجربی با فرمول تئوری $1/M$. سمت راست: افزایش خطی زمان اجرا به موازات افزایش هدررفت نمونه‌ها.}
		\label{fig:acceptance-rate-runtime}
	\end{figure}
	
	\textbf{تحلیل نرخ پذیرش \lr{(Acceptance Rate)}:}
	همان‌طور که در جدول و نمودار سمت چپ مشاهده می‌شود، در بهینه‌ترین حالت، یعنی $M=2.9921$، تنها حدود $33\%$ نمونه‌ها پذیرفته می‌شوند. با بازتر کردن کران‌ها و رسیدن به $M \approx 9.77$، نرخ پذیرش به حدود $10.2\%$ سقوط می‌کند. تطابق کامل نقاط آبی تجربی و قرمز تئوری در نمودار، نشان‌دهنده رفتار کاملاً و تصادفی و مستقل در الگوریتم پیاده‌سازی شده است. در مدل کاری ما، یعنی $M=4.1891$، نرخ پذیرش حدود $23.8\%$ است؛ این یعنی برای تولید ۱۰۰ هزار داده معتبر، الگوریتم مجبور به تولید و پردازش بیش از ۴۲۰ هزار نمونه پیشنهادی شده است که معادل $76\%$ هدررفت محاسباتی است.
	
	\textbf{تحلیل پیچیدگی زمانی \lr{(Runtime Complexity)}:}
	نمودار سمت راست رابطه مستقیم بین افزایش $M$ و زمان اجرای الگوریتم را نشان می‌دهد. از منظر پیچیدگی محاسباتی، برای استخراج $N$ نمونه موفق، حلقه الگوریتم باید به طور متوسط $N \times M$ بار تکرار شود. بنابراین زمان اجرا از اردر $\mathcal{O}(M)$ است و روندی صعودی دارد.
	
	\textit{نکته تحلیلی:} نوسانات و افت و خیزهای جزئی در نمودار زمان، مثلاً کاهش زمان در $M \approx 3.5$، یک پدیده رایج در ارزیابی‌های سطح میکرو \lr{(Micro-benchmarking)} است. از آنجا که زمان‌ها در مقیاس هزارم ثانیه، یعنی میلی‌ثانیه، ثبت شده‌اند، این نوسانات ناشی از زمان‌بندی سیستم‌عامل \lr{(OS Scheduling)}، وضعیت حافظه پنهان پردازنده \lr{(CPU Cache)} و فرآیندهای پس‌زمینه \lr{(Background Processes)} است؛ اما روند کلی \lr{(Macro-trend)} کاملاً صعودی و منطبق بر تحلیل ریاضی الگوریتم است.
	
	\textbf{نتیجه‌گیری برای ورود به فاز بعدی:}
	الگوریتم پذیرش-رد برای تولید داده‌های کلی موفقیت‌آمیز بود، اما نشان داد که به دلیل ماهیت رد کردن نمونه‌ها، هزینه‌بر است. حال اگر بخواهیم یک رویداد نادر با احتمال $10^{-4}$، مانند عبور کلرامین از آستانه $\tau=13.0089$، را صرفاً با این داده‌های تولید شده تخمین بزنیم، یعنی روش مونت‌کارلو استاندارد، نیازمند تولید میلیون‌ها نمونه هستیم که با توجه به هدررفت $76$ درصدی این الگوریتم، به گلوگاه محاسباتی شدیدی برخورد خواهیم کرد. این دقیقاً همان دلیلی است که نیاز به روش‌های پیشرفته‌تری مانند \lr{Importance Sampling} را ایجاب می‌کند.
	
	\section{تخمین مونت‌کارلو استاندارد \lr{(Standard Monte Carlo)} و چالش‌های رویداد نادر}
	
	پس از مدل‌سازی داده‌ها، یعنی $\mu=7.1223$ و $\sigma=1.5828$، و تعیین آستانه خطر، یعنی $\tau=13.0089$، هدف اصلی پروژه تخمین احتمال وقوع رویداد نادر
	\[
	\gamma = P(X > \tau)
	\]
	است که مقدار واقعی آن برابر با $10^{-4}$ می‌باشد.
	
	طبیعی‌ترین و ساده‌ترین رویکرد برای این کار، استفاده از روش مونت‌کارلو استاندارد \lr{(Crude Monte Carlo)} است. در این روش، $N$ نمونه مستقل $X_1,\dots,X_N$ از توزیع هدف $f(x)$، با استفاده از الگوریتم باکس-مولر، تولید شده و احتمال $\gamma$ به کمک میانگین‌گیری از تابع نشانگر \lr{(Indicator Function)} تخمین زده می‌شود:
	\[
	\hat{\gamma}_{MC} =
	\frac{1}{N}
	\sum_{i=1}^{N}
	\mathbb{I}(X_i > \tau)
	\]
	
	از نظر ریاضی، این تخمین‌گر \textbf{نااریب \lr{(Unbiased)}} است، زیرا:
	\[
	\mathbb{E}[\hat{\gamma}_{MC}] = \gamma
	\]
	با این حال، در مواجهه با رویدادهای نادر، مسئله اصلی نااریب بودن نیست، بلکه \textbf{واریانس \lr{(Variance)}} تخمین‌گر است.
	
\subsection{ارزیابی جامع عملکرد و محدودیت‌های مونت‌کارلو استاندارد}

در روش مونت‌کارلو استاندارد (SMC)، از آنجا که هر رویداد یک متغیر برنولی با احتمال موفقیت $\gamma$ است، واریانس تخمین‌گر $\hat{\gamma}_{MC}$ برابر است با:
\[
\operatorname{Var}(\hat{\gamma}_{MC}) = \frac{\gamma(1-\gamma)}{N}
\]
اما در ارزیابی رویدادهای نادر، معیاری که اهمیت حیاتی دارد، \textbf{خطای نسبی \lr{(Relative Error)}} است که به صورت ضریب تغییرات \lr{(Coefficient of Variation)} تعریف می‌شود:
\[
\text{Relative Error} =
\frac{\sqrt{\operatorname{Var}(\hat{\gamma}_{MC})}}{\gamma} =
\sqrt{\frac{1-\gamma}{N\gamma}} \approx \frac{1}{\sqrt{N\gamma}}
\]
این رابطه ریاضی، هسته‌ی اصلی چالش در تخمین رویدادهای نادر با روش مونت‌کارلو را آشکار می‌سازد. نتایج شبیه‌سازی ارائه شده در جدول، دقیقاً این بحران را به تصویر می‌کشد:

\begin{itemize}
	\item \textbf{در $N = 10^3$ (فاجعه نمونه‌برداری):} مخرج کسر خطای نسبی برابر با $\sqrt{1000 \times 10^{-4}} = \sqrt{0.1} \approx 0.316$ است که منجر به خطای نسبی خردکننده حدود $3.16$ می‌شود. مهم‌تر از آن، در \textbf{$92.6\%$} از اجراها، تخمین‌گر مقدار دقیقاً \textbf{صفر} را برمی‌گرداند. این پدیده که به آن \lr{Zero-Estimate Phenomenon} می‌گویند، با تقریب پواسون قابل توضیح است: امید ریاضی تعداد رویدادها $N\gamma = 0.1$ است؛ طبق توزیع پواسون، احتمال مشاهده صفر رویداد برابر با $e^{-0.1} \approx 0.904$ است که با نتیجه تجربی همخوانی دارد. یک تخمین صفر در مهندسی قابلیت اطمینان به معنای غیرممکن بودن خرابی است، که نتیجه‌ای کاملاً گمراه‌کننده و خطرناک است.
	\item \textbf{در $N = 10^4$:} هنوز در $35.5\%$ از آزمایش‌ها هیچ رویداد نادری مشاهده نمی‌شود و خطای نسبی $1.00$ (معادل $100\%$) نشان‌دهنده عدم قطعیت شدید است.
	\item \textbf{پایداری در $N \ge 10^5$:} تنها زمانی که حجم نمونه به صدها هزار تا یک میلیون می‌رسد (یعنی $N\gamma \gg 1$)، تخمین‌گر پایدار شده و خطای نسبی به زیر $10\%$ کاهش می‌یابد.
\end{itemize}

\subsubsection*{تحلیل نرخ همگرایی و تایید تئوری (پاسخ به پرسش تحلیلی سوم)}\label{sec:q3}

پرسش کلیدی این است: روش مونت‌کارلو با چه نرخی همگرا می‌شود و آیا می‌توان این همگرایی را تسریع کرد؟ نمودار سمت راست در شکل~\ref{fig:mc-evaluation-combined} به این پرسش پاسخ می‌دهد. طبق تئوری، خطای استاندارد (SE) متناسب با $N^{-1/2}$ است:
\[
\log(\text{SE}) \approx \log(\sqrt{\gamma}) - \frac{1}{2}\log(N)
\]
این معادله، یک خط راست با \textbf{شیب دقیقاً $-0.5$} در نمودار \lr{Log-Log} را پیش‌بینی می‌کند. همان‌طور که در عنوان نمودار و محاسبات نوت‌بوک نشان داده شده، شیب تجربی برازش داده شده برابر با \textbf{$-0.50$} است که با دقت بسیار بالایی تئوری را تایید می‌کند.

\textbf{تفسیر مهندسی این شیب:} این شیب بیانگر نفرین مونت‌کارلو \lr{(Curse of Monte Carlo)} است. این نرخ نشان می‌دهد که خطای محاسباتی با ریشه دوم حجم نمونه‌ها کاهش می‌یابد. به زبان ساده‌تر:
\begin{itemize}
	\item برای \textbf{نصف کردن} خطا، باید حجم محاسبات را \textbf{۴ برابر} کنیم.
	\item برای افزودن \textbf{یک رقم اعشار دقت} (کاهش ۱۰ برابری خطا)، باید حجم محاسبات را \textbf{۱۰۰ برابر} کنیم.
\end{itemize}

\textbf{نتیجه‌گیری برای ورود به فاز نهایی:} تحلیل این بخش ثابت کرد که اگرچه مونت‌کارلو استاندارد از نظر تئوری صحیح و نااریب است (Unbiased)، اما برای رویدادهایی با احتمال $10^{-4}$ و کمتر، به شدت ناکارآمد است. مشکل، شیب $ -0.5$ نیست، بلکه بزرگ بودن ثابت تناسب در رابطه خطای نسبی ($1/\sqrt{\gamma}$) است. این ناکارآمدی، انگیزه اصلی برای استفاده از روش \textbf{نمونه‌برداری اهمیتی \lr{(Importance Sampling)}} در بخش بعدی است؛ روشی که قصد دارد با تغییر هوشمندانه توزیع نمونه‌برداری، این ثابت را به شدت کاهش دهد و تخمینی پایدار با حجم نمونه بسیار کمتر ارائه دهد.


	\section{نمونه‌برداری اهمیتی \lr{(Importance Sampling)} و شکستن سد واریانس}
	
	همان‌طور که در بخش قبل مشاهده شد، روش مونت‌کارلو استاندارد \lr{(Crude MC)} اکثر توان محاسباتی خود را در بدنه اصلی توزیع، جایی که تابع نشانگر $\mathbb{I}(X > \tau) = 0$ است، هدر می‌دهد. برای حل این مشکل، از تکنیک \textbf{نمونه‌برداری اهمیتی \lr{(IS)}} استفاده می‌کنیم. ایده اصلی این روش، ریشه در مفهوم ریاضی تغییر اندازه \lr{(Change of Measure)} دارد.
	
	ما نمونه‌ها را به جای توزیع اصلی $f(x)$، از یک توزیع پیشنهادی $q(x)$ تولید می‌کنیم که مرکز جرم آن به سمت ناحیه رویداد نادر، یعنی آستانه $\tau$، منتقل شده است. برای حفظ خاصیت نااریب بودن تخمین‌گر، هر نمونه با ضریب اهمیت \lr{(Importance Weight)} معادل $w(x) = f(x)/q(x)$ اصلاح می‌شود:
	\[
	\gamma =
	\mathbb{E}_f[\mathbb{I}(X>\tau)]
	=
	\mathbb{E}_q
	\left[
	\mathbb{I}(X>\tau)
	\frac{f(X)}{q(X)}
	\right]
	\]
	\[
	\hat{\gamma}_{IS} =
	\frac{1}{N}
	\sum_{i=1}^{N}
	\mathbb{I}(X_i > \tau) w(X_i),
	\qquad
	X_i \sim q
	\]
	
	\textbf{طراحی توزیع پیشنهادی:}
	در این مدل، $q(x)$ یک توزیع نرمال انتخاب شد که میانگین آن دقیقاً روی آستانه خطر تنظیم شده است، یعنی $\mu_q = \tau = 13.0089$، و انحراف معیار آن برابر با توزیع اصلی است، یعنی $\sigma_q = 1.5828$. این تغییر مکان باعث می‌شود تقریباً نیمی از نمونه‌های تولید شده در ناحیه بحرانی، یعنی $X > \tau$، قرار گیرند.
	
	\textbf{نتایج شگفت‌انگیز کاهش واریانس:}
	جداول خروجی شبیه‌سازی، قدرت بی‌نظیر این روش را اثبات می‌کنند:
	
	\begin{itemize}
		\item \textbf{حل مشکل تخمین صفر:} در حجم نمونه $N=10^4$، در حالی که روش \lr{MC} در $35.5\%$ مواقع هیچ رویدادی را ثبت نمی‌کرد، یعنی $\hat{\gamma} = 0$، روش \lr{IS} به دلیل نمونه‌برداری مستقیم از ناحیه بحرانی، نرخ تخمین صفر را به \textbf{$0\%$} رساند.
		\item \textbf{ضریب کاهش واریانس \lr{(Variance Reduction Factor)}:} واریانس تخمین‌گر \lr{IS} بین \textbf{$2256$ تا $3202$ برابر} کمتر از \lr{MC} است.
		\item \textbf{کاهش خطای نسبی:} در $N=10^4$، خطای نسبی از $102.6\%$ در روش \lr{MC} به تنها \textbf{$2.0\%$} در روش \lr{IS} کاهش یافته است. به عبارت دیگر، دقت روش \lr{IS} با $10^4$ نمونه، معادل دقت روش \lr{MC} با بیش از $25 \times 10^6$ نمونه است.
	\end{itemize}
	
	\subsection{تحلیل بصری عملکرد \lr{Importance Sampling} در برابر \lr{Monte Carlo}}
	
	برای ارزیابی دقیق‌تر و درک شهودی برتری \lr{IS}، نمودار سه‌گانه‌ای رسم شده است که سه جنبه کلیدی را مقایسه می‌کند:
	
	\begin{figure}[H]
		\centering
		\includegraphics[width=0.95\textwidth]{../results/04_importance_sampling.png}
		\caption{مقایسه سه‌گانه عملکرد \lr{Importance Sampling} و \lr{Monte Carlo}: همگرایی خطا، هیستوگرام وزن‌ها و پایداری تخمین در حجم نمونه‌های مختلف.}
		\label{fig:is-vs-mc-triple}
	\end{figure}
	
	\begin{enumerate}
		\item \textbf{همگرایی خطا:} نمودار \lr{Log-Log} خطای استاندارد نشان می‌دهد که هر دو روش از نرخ همگرایی تئوری $\mathcal{O}(N^{-1/2})$ پیروی می‌کنند؛ شیب برای \lr{MC} برابر $-0.49$ و برای \lr{IS} برابر $-0.53$ است. با این حال، انتقال توزیع توانسته خطای روش \lr{IS} را حدود دو الی سه مرتبه بزرگی \lr{(Order of Magnitude)} نسبت به \lr{MC} کاهش دهد.
		\item \textbf{توزیع وزن‌های اهمیت:} این هیستوگرام پایداری وزن‌ها را در ناحیه نادر نشان می‌دهد. وزن‌ها مقادیر بسیار کوچکی دارند، با میانگین تقریباً $2.00 \times 10^{-4}$، و هیچ نمونه منفردی با وزن نامتعارف که منجر به اختلال در میانگین‌گیری شود، وجود ندارد.
		\item \textbf{پایداری تخمین در حجم نمونه پایین:} این نمودار شکست \lr{MC} در مواجهه با رویداد نادر را به تصویر می‌کشد؛ در $N=10^3$، نوارهای خطای \lr{MC} به قدری بزرگ است که به اعداد منفی می‌رسد. در مقابل، تخمین \lr{IS} حتی در این حجم نمونه کوچک، با دقتی بالا و نوارهای خطای بسیار محدود روی احتمال واقعی، یعنی $\gamma = 1.0 \times 10^{-4}$، منطبق است.
	\end{enumerate}
	
	\section{تحلیل پایداری وزن‌ها و ضخامت دم توزیع}
	
	\textbf{پاسخ به پرسش تحلیلی دوم}\label{sec:q2}
	
	همان‌طور که در نمودار میانی بخش قبل مشاهده شد، پایداری وزن‌ها رمز موفقیت \lr{IS} است. یکی از خطرات پنهان در این روش، پدیده \textbf{انفجار واریانس \lr{(Variance Explosion)}} به دلیل ناپایداری وزن‌هاست.
	
	\textbf{قانون طلایی در \lr{IS}:}
	\textit{توزیع پیشنهادی $q(x)$ باید در ناحیه مورد نظر، دمی ضخیم‌تر \lr{(Heavier Tail)} یا حداقل برابر با دم توزیع هدف $f(x)$ داشته باشد.}
	
	دلیل ریاضی این امر در کسر $w(x) = \frac{f(x)}{q(x)}$ نهفته است. اگر $q(x)$ دمی سبک‌تر از $f(x)$ داشته باشد، مثلاً $\sigma_q = 0.5\sigma$، با حرکت به سمت بی‌نهایت، مخرج کسر بسیار سریع‌تر از صورت به صفر میل می‌کند و در نتیجه:
	\[
	\lim_{x \to \infty} w(x) = \infty
	\]
	این یعنی اگر بر حسب تصادف نمونه‌ای در انتهای دم تولید شود، وزنی نجومی دریافت می‌کند که کل میانگین را مختل کرده و واریانس را منفجر می‌سازد.
	
	برای کمی‌سازی این پایداری، از معیار \textbf{اندازه نمونه مؤثر \lr{(Effective Sample Size - ESS)}} استفاده می‌شود:
	\[
	ESS =
	\frac{\left(\sum w_i\right)^2}{\sum w_i^2}
	\]
	اگر وزن‌ها یکنواخت باشند، $ESS \approx N$ است، و اگر یک نمونه وزن بی‌نهایت بگیرد، $ESS \to 1$ خواهد بود.
	
	همان‌طور که در شکل \ref{fig:weight-tails-analysis} مشاهده می‌شود، انتخاب توزیع پیشنهادی با دم سبک‌تر (نمودار قرمز)، منجر به رشد نمایی تابع وزن شده و مقادیر لگاریتم وزن‌ها را به اعداد نجومی (تا ابعاد $10^{60}$) می‌رساند. این پدیده مستقیماً باعث انفجار گشتاور دوم و بی‌نهایت شدن واریانس تخمین‌گر می‌گردد. در نقطه مقابل، توزیع با دم سنگین‌تر (نمودار سبز) رفتاری کاملاً کراندار و ایمن از خود نشان می‌دهد.

	\centering
	\includegraphics[width=0.95\textwidth]{../results/04_weight_tails_Q2.png}
	\caption{تحلیل اثر ضخامت دم توزیع پیشنهادی؛ از چپ به راست: رشد نمایی تابع وزن در دم سبک‌تر، گستردگی نجومی در هیستوگرام لگاریتم وزن‌ها، و نمودار صعودی گشتاور دوم که نشان‌دهنده انفجار واریانس است (در مقایسه با رفتار کراندار و پایدار دم سنگین‌تر).}

این تفاوت در شکل دم‌ها، خودش را در معیار اندازه نمونه مؤثر، یعنی \lr{ESS}، نشان می‌دهد. وقتی توزیع پیشنهادی دم سبک‌تری دارد ($\sigma_q = 0.5\sigma$)، فقط تعداد کمی از نمونه‌ها وزن‌های بسیار بزرگی می‌گیرند و عملاً بیشتر نمونه‌ها نقش چندانی در تخمین ندارند؛ همان چیزی که در هیستوگرام نیز دیده می‌شود. به همین دلیل مقدار \lr{ESS} به شدت کاهش پیدا می‌کند و نسبت $\text{ESS}/N$ به مقدار بسیار کوچک $0.0002$ می‌رسد. این عدد نشان می‌دهد که تخمین در این حالت تقریباً ناپایدار شده و وزن‌ها کنترل خود را از دست داده‌اند.

در مقابل، وقتی توزیع پیشنهادی دم سنگین‌تری دارد ($\sigma_q = 1.5\sigma$)، احتمال دیدن نمونه‌های دور از مرکز بیشتر می‌شود و در نتیجه مخرج وزن، یعنی $q(x)$، برای این نقاط بیش از حد کوچک نمی‌شود. بنابراین وزن‌ها رفتار متعادل‌تری دارند و نسبت $\text{ESS}/N$ به مقدار بهترِ $0.0170$ می‌رسد. کنار هم گذاشتن این نتایج عددی و شکل هیستوگرام نشان می‌دهد که برای داشتن یک تخمین پایدار، توزیع پیشنهادی باید دم‌هایی پهن‌تر از توزیع هدف داشته باشد؛ در غیر این صورت، چند وزن بزرگ می‌توانند کل تخمین را تحت تأثیر قرار دهند.


	\section{بهینه‌سازی پارامتر انتقال و دلایل شکست \lr{IS}}
	
	\textbf{پاسخ به پرسش تحلیلی چهارم}\label{sec:q4}
	
	آیا در روش \lr{IS} همواره موفقیت تضمین شده است؟ خیر. انتخاب بد میانگین توزیع پیشنهادی، یعنی $\mu_q$، می‌تواند عملکرد \lr{IS} را حتی از مونت‌کارلوی استاندارد بدتر کند.
	
	برای درک این موضوع، خطای نسبی روش \lr{IS} به ازای مقادیر مختلف $\mu_q$ در یک نمودار ترسیم شده است که شکل یک منحنی \textbf{U-شکل \lr{(U-shaped)}} را نشان می‌دهد:
	
	\begin{enumerate}
		\item \textbf{انتقال ناکافی \lr{(Under-shifting)}، $\mu_q \approx \mu$:} اگر $\mu_q$ نزدیک به میانگین اصلی، یعنی $7.12$، بماند، توزیع پیشنهادی تفاوت چندانی با توزیع اصلی ندارد. در این حالت همچنان اکثر نمونه‌ها خارج از ناحیه نادر می‌افتند و به دلیل وجود نویز حاصل از ضرب شدن وزن‌ها، واریانس تخمین‌گر حتی ممکن است کمی بیشتر از \lr{MC} استاندارد شود.
		\item \textbf{انتقال بیش از حد \lr{(Over-shifting)}، $\mu_q \gg \tau$:} اگر توزیع را خیلی به جلو ببریم، مثلاً $\mu_q = 19$، تقریباً تمام نمونه‌ها بزرگتر از $\tau$ می‌شوند، اما چون از بدنه $f(x)$ بسیار دور شده‌اند، وزن آن‌ها، یعنی $f(x)/q(x)$، به شدت به صفر میل می‌کند. در این میان، معدود نمونه‌هایی که در سمت چپ این توزیع و نزدیک به $\tau$ می‌افتند، وزن‌های نسبتاً بسیار بزرگی خواهند داشت. این عدم تقارن شدید باعث افت شدید \lr{ESS} و افزایش چشمگیر خطای نسبی در سمت راست نمودار U-شکل می‌شود.
		\item \textbf{نقطه بهینه \lr{(Sweet Spot)}:} بهترین عملکرد دقیقاً در مجاورت آستانه $\tau$ رخ می‌دهد. در اینجا $\mu_q \approx 14.25$ بهترین نتیجه را با خطای نسبی $0.014$ ثبت کرده است. در این ناحیه، هم تعداد نمونه‌های موفق بالاست و هم توزیع وزن‌ها متوازن و پایدار است.
	\end{enumerate}

	\section{جمع‌بندی و نتیجه‌گیری نهایی پروژه}
	
	این پروژه یک مسیر مهندسی و آماری کامل را برای ارزیابی رویدادهای نادر در کیفیت آب، با تمرکز بر متغیر \lr{Chloramines}، با موفقیت طی کرد:
	
	\begin{enumerate}
		\item \textbf{مدل‌سازی آماری:} در ابتدا با استفاده از برازش \lr{MLE}، توزیع نرمال به عنوان بهترین مدل توصیف‌کننده داده‌ها شناسایی شد و احتمال آستانه بحرانی $\tau = 13.0089$ برابر با مقدار نادر $10^{-4}$ تعیین گردید.
		\item \textbf{تولید تصادفی پایه:} الگوریتم \lr{Accept-Reject} به صورت بنیادین پیاده‌سازی شد. اگرچه این روش برای تولید داده‌های منطبق بر توزیع موفقیت‌آمیز بود، اما نشان داد که برای کاوش در دم توزیع به شدت ناکارآمد است.
		\item \textbf{چالش مونت‌کارلو استاندارد:} شبیه‌سازی نشان داد که روش \lr{Crude MC} از پدیده تخمین صفر رنج می‌برد و برای رسیدن به خطای قابل قبول نیازمند میلیون‌ها نمونه‌برداری است؛ موضوعی که نرخ همگرایی کند $\mathcal{O}(N^{-1/2})$ و واریانس بالای آن را به اثبات رساند.
		\item \textbf{تکنیک پیشرفته کاهش واریانس:} در نهایت، با استفاده از الگوریتم نمونه‌برداری اهمیتی \lr{(Importance Sampling)}، یک تغییر پارادایم از افزایش حجم نمونه به تغییر هوشمندانه توزیع انجام شد. با انتخاب دقیق $\mu_q \approx \tau$ و رعایت شرط ضخامت دم توزیع پیشنهادی، واریانس مسئله بیش از $2500$ برابر کاهش یافت و تخمین دقیق رویداد نادر با حجم محاسباتی بسیار اندک ممکن گردید.
	\end{enumerate}
